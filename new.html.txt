    <script>
        // Music Control Functionality
        const songSelect = document.getElementById('songSelect');
        const playPauseBtn = document.getElementById('playPauseBtn');
        const bgMusic = document.getElementById('bgMusic');
        let isPlaying = false;

        songSelect.addEventListener('change', function() {
            if (this.value) {
                bgMusic.src = this.value;
                // When a song is selected, automatically play it
                bgMusic.play();
                isPlaying = true;
                playPauseBtn.textContent = '‚è∏Ô∏è Pause Music';
            }
        });

        playPauseBtn.addEventListener('click', function() {
            if (!bgMusic.src) {
                alert('Please select a song first! üéµ');
                return;
            }

            if (isPlaying) {
                bgMusic.pause();
                playPauseBtn.textContent = '‚ñ∂Ô∏è Play Music';
            } else {
                bgMusic.play();
                playPauseBtn.textContent = '‚è∏Ô∏è Pause Music';
            }
            isPlaying = !isPlaying;
        });

        // Sparkle effect on mouse move
        document.addEventListener('mousemove', function(e) {
            if (Math.random() > 0.9) { // Create sparkles randomly
                const sparkle = document.createElement('div');
                sparkle.className = 'sparkle';
                sparkle.style.left = e.pageX + 'px';
                sparkle.style.top = e.pageY + 'px';
                document.body.appendChild(sparkle);

                setTimeout(() => {
                    sparkle.remove();
                }, 1000);
            }
        });

        // Parallax effect for hearts
        document.addEventListener('mousemove', function(e) {
            const hearts = document.querySelectorAll('.heart');
            const x = e.clientX / window.innerWidth;
            const y = e.clientY / window.innerHeight;

            hearts.forEach((heart, index) => {
                const speed = (index + 1) * 0.5;
                const xMove = (x - 0.5) * speed * 20;
                const yMove = (y - 0.5) * speed * 20;
                heart.style.transform = `translate(${xMove}px, ${yMove}px)`;
            });
        });

        // Add smooth scroll behavior
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                e.preventDefault();
                const target = document.querySelector(this.getAttribute('href'));
                if (target) {
                    target.scrollIntoView({
                        behavior: 'smooth'
                    });
                }
            });
        });
    </script>
